  <!-- JAVASCRIPT LOGIC -->
  <script>
    // Data bawaan awal (jika local storage kosong)
    const DEFAULT_PRODUCTS = [
      { id: '1', sku: 'WXL-SHIRT-01', brand: 'Wixel Wear', letak: 'Gantungan A', lantai: 'Lantai 1' },
      { id: '2', sku: 'NKE-RUN-99', brand: 'Nike', letak: 'Rak B', lantai: 'Lantai 3' },
      { id: '3', sku: 'ZRA-JKT-05', brand: 'Zara', letak: 'Gantungan B', lantai: 'Lantai 1' },
      { id: '4', sku: 'ADD-SLIP-12', brand: 'Adidas', letak: 'Rak A', lantai: 'Lantai 2' },
      { id: '5', sku: 'UNI-PNT-44', brand: 'Uniqlo', letak: 'Gantungan A', lantai: 'Lantai 2' },
      { id: '6', sku: 'HM-HOOD-08', brand: 'H&M', letak: 'Gantungan B', lantai: 'Lantai 2' },
      { id: '7', sku: 'PMA-SNEAK-21', brand: 'Puma', letak: 'Rak B', lantai: 'Lantai 1' }
    ];

    // State Aplikasi
    let products = JSON.parse(localStorage.getItem('gudang_products')) || DEFAULT_PRODUCTS;
    let selectedProductId = null;
    let currentSearchTerm = '';

    // Elemen DOM
    const tableBody = document.getElementById('product-table-body');
    const tableSummary = document.getElementById('table-summary');
    const searchInput = document.getElementById('search-input');
    const clearSearchBtn = document.getElementById('clear-search-btn');
    const addFormContainer = document.getElementById('add-form-container');
    const selectedItemCard = document.getElementById('selected-item-card');

    // Simpan Data ke LocalStorage
    function saveToStorage() {
      localStorage.setItem('gudang_products', JSON.stringify(products));
    }

    // Tampilkan Toast Notifikasi
    function showToast(message, type = 'success') {
      const container = document.getElementById('toast-container');
      const box = document.getElementById('toast-box');
      const msg = document.getElementById('toast-message');

      msg.innerText = message;
      
      // Bersihkan kelas warna sebelumnya
      box.className = "px-4 py-3 rounded-xl shadow-xl border text-sm font-bold flex items-center gap-2 ";
      
      if (type === 'success') {
        box.classList.add('bg-emerald-500', 'text-white', 'border-emerald-600');
      } else if (type === 'error') {
        box.classList.add('bg-rose-500', 'text-white', 'border-rose-600');
      } else {
        box.classList.add('bg-slate-800', 'text-white', 'border-slate-700');
      }

      container.classList.remove('hidden');
      setTimeout(() => {
        container.classList.add('hidden');
      }, 3000);
    }

    // Toggle Tampilan Form Tambah Produk
    function toggleAddForm() {
      addFormContainer.classList.toggle('hidden');
    }

    // Carian Produk
    function handleSearch(term) {
      currentSearchTerm = term.trim().toLowerCase();
      if (currentSearchTerm) {
        clearSearchBtn.classList.remove('hidden');
      } else {
        clearSearchBtn.classList.add('hidden');
      }
      render();
    }

    function clearSearch() {
      searchInput.value = '';
      currentSearchTerm = '';
      clearSearchBtn.classList.add('hidden');
      render();
    }

    // Pilih Produk untuk Ditampilkan Di Peta
    function selectProduct(id) {
      selectedProductId = id;
      const prod = products.find(p => p.id === id);
      if (prod) {
        showToast(`Melihat lokasi untuk ${prod.sku} (${prod.brand})`, 'info');
      }
      render();
    }

    // Batal Pilih Produk
    function deselectProduct() {
      selectedProductId = null;
      render();
    }

    // Hapus Produk dari Daftar
    function deleteProduct(id) {
      products = products.filter(p => p.id !== id);
      if (selectedProductId === id) {
        selectedProductId = null;
      }
      saveToStorage();
      showToast('Produk berjaya dipadam dari sistem', 'info');
      render();
    }

    // Submit Form Tambah Produk Baru
    function handleFormSubmit(event) {
      event.preventDefault();
      
      const skuInput = document.getElementById('input-sku');
      const brandInput = document.getElementById('input-brand');
      const letakSelect = document.getElementById('input-letak');
      const lantaiSelect = document.getElementById('input-lantai');

      const sku = skuInput.value.toUpperCase().trim();
      const brand = brandInput.value.trim();
      const letak = letakSelect.value;
      const lantai = lantaiSelect.value;

      if (!sku || !brand) {
        showToast('Sila isi semua ruangan!', 'error');
        return;
      }

      const newProduct = {
        id: Date.now().toString(),
        sku,
        brand,
        letak,
        lantai
      };

      products.unshift(newProduct);
      saveToStorage();

      // Reset Form & Tutup
      skuInput.value = '';
      brandInput.value = '';
      letakSelect.value = 'Rak A';
      lantaiSelect.value = 'Lantai 1';
      toggleAddForm();

      showToast('Produk berjaya didaftarkan ke lokasi!', 'success');
      render();
    }

    // Fungsi Utama Merender Seluruh Perubahan UI
    function render() {
      // 1. Filter Produk berdasarkan Search
      const filtered = products.filter(product => {
        return (
          product.sku.toLowerCase().includes(currentSearchTerm) ||
          product.brand.toLowerCase().includes(currentSearchTerm) ||
          product.letak.toLowerCase().includes(currentSearchTerm) ||
          product.lantai.toLowerCase().includes(currentSearchTerm)
        );
      });

      // Update Teks Ringkasan Jadual
      tableSummary.innerText = `Menunjukkan ${filtered.length} rekod barang`;

      // 2. Render Jadual HTML
      tableBody.innerHTML = '';
      if (filtered.length === 0) {
        tableBody.innerHTML = `
          <tr>
            <td colspan="5" class="py-12 text-center text-slate-400">
              <p class="font-bold text-xs">Tiada data dijumpai</p>
              <p class="text-[10px] mt-1">Gunakan kotak carian atau daftar produk baru.</p>
            </td>
          </tr>
        `;
      } else {
        filtered.forEach(prod => {
          const isHighlighted = selectedProductId === prod.id;
          const letakBadgeClass = prod.letak.startsWith('Rak') 
            ? 'bg-amber-50 text-amber-700' 
            : 'bg-indigo-50 text-indigo-700';

          tableBody.innerHTML += `
            <tr class="hover:bg-slate-50/50 transition duration-100 ${isHighlighted ? 'bg-blue-50/30' : ''}">
              <td class="py-3.5 px-5 font-bold text-xs text-blue-700 tracking-wider">${prod.sku}</td>
              <td class="py-3.5 px-5 text-slate-700 font-semibold text-xs">${prod.brand}</td>
              <td class="py-3.5 px-5">
                <span class="text-[10px] font-bold px-2.5 py-1 rounded-full ${letakBadgeClass}">
                  ${prod.letak}
                </span>
              </td>
              <td class="py-3.5 px-5 text-xs text-slate-500 font-medium">${prod.lantai}</td>
              <td class="py-3.5 px-5">
                <div class="flex items-center justify-center gap-2">
                  <button
                    onclick="selectProduct('${prod.id}')"
                    class="px-3 py-1.5 bg-blue-50 hover:bg-blue-100 text-blue-700 text-[10px] font-bold rounded-lg transition"
                  >
                    Tunjuk Peta
                  </button>
                  <button
                    onclick="deleteProduct('${prod.id}')"
                    class="p-1.5 text-slate-400 hover:text-rose-600 hover:bg-rose-50 rounded-lg transition"
                    title="Hapus Produk"
                  >
                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                      <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                    </svg>
                  </button>
                </div>
              </td>
            </tr>
          `;
        });
      }

      // 3. Hitung Jumlah Item di Setiap Lokasi Peta
      const counts = {
        'Gantungan A': 0,
        'Gantungan B': 0,
        'Rak A': 0,
        'Rak B': 0
      };
      products.forEach(p => {
        if (counts[p.letak] !== undefined) {
          counts[p.letak]++;
        }
      });

      // Update Angka di Peta
      document.getElementById('count-Gantungan-A').innerText = `${counts['Gantungan A']} SKU tersimpan`;
      document.getElementById('count-Gantungan-B').innerText = `${counts['Gantungan B']} SKU tersimpan`;
      document.getElementById('count-Rak-A').innerText = `${counts['Rak A']} SKU tersimpan`;
      document.getElementById('count-Rak-B').innerText = `${counts['Rak B']} SKU tersimpan`;

      // 4. Update Tampilan Glow Map (Jika ada produk yang terpilih)
      const selectedProduct = products.find(p => p.id === selectedProductId);
      
      // Reset semua style peta ke normal
      const mapKeys = ['Gantungan-A', 'Gantungan-B', 'Rak-A', 'Rak-B'];
      mapKeys.forEach(key => {
        const el = document.getElementById(`map-loc-${key}`);
        el.className = "bg-white border-slate-200 p-4 rounded-xl border-2 flex flex-col items-center justify-center text-center transition-all duration-200";
      });

      if (selectedProduct) {
        // Tampilkan Detail Card di Peta
        selectedItemCard.classList.remove('hidden');
        document.getElementById('card-brand').innerText = `Brand: ${selectedProduct.brand}`;
        document.getElementById('card-sku').innerText = selectedProduct.sku;
        document.getElementById('card-letak').innerText = selectedProduct.letak;
        document.getElementById('card-lantai').innerText = selectedProduct.lantai;

        // Beri efek glow biru pada lokasi terpilih
        const targetMapId = `map-loc-${selectedProduct.letak.replace(' ', '-')}`;
        const targetEl = document.getElementById(targetMapId);
        if (targetEl) {
          targetEl.className = "p-4 rounded-xl border-2 flex flex-col items-center justify-center text-center transition-all duration-200 bg-blue-600 text-white border-blue-700 shadow-lg scale-105 animate-pulse";
        }

        selectedItemCard.classList.add('hidden');
